## ğŸ’¡ Usage Examples

### Basic Translation
```typescript
import { useTranslation } from '../hooks/useTranslation';

function MyComponent() {
  const { t } = useTranslation();
  
  return <h1>{t('app.title')}</h1>;
  // German: "BoxCopilot"
  // English: "BoxCopilot"
}
```

### Variable Interpolation
```typescript
const { t } = useTranslation();

// With numbers
t('boxes.boxNumber', { number: 42 })
// German: "Box #42"
// English: "Box #42"

// With counts
t('items.itemsCount', { count: 5 })
// German: "5 Items"
// English: "5 Items"
```

### Changing Language
```typescript
const { setLanguage } = useTranslation();

// Switch to English
setLanguage('en');

// Switch to German
setLanguage('de');
```

### Checking Current Language
```typescript
const { language } = useTranslation();

if (language === 'de') {
  // German-specific logic
} else if (language === 'en') {
  // English-specific logic
}
```
